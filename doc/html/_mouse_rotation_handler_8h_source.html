102  void applyTransform(const defaulttype::Vector3 & phi) {
103  PointCloud::Ptr
104  cloud (d_in.getValue().getPointCloud()),
105  cloud_tf_1 (new PointCloud),
106  cloud_tf_2 (new PointCloud);
107 
108  Eigen::Affine3f transform (Eigen::Affine3f::Identity());
109  Eigen::Matrix3f rotation (
110  Eigen::AngleAxisf((phi[0]*M_PI) / 180, Eigen::Vector3f::UnitX()) *
111  Eigen::AngleAxisf((phi[1]*M_PI) / 180, Eigen::Vector3f::UnitY()) *
112  Eigen::AngleAxisf((phi[2]*M_PI) / 180, Eigen::Vector3f::UnitZ())
113  );
114  transform.rotate(rotation);
115 
116  pcl::transformPointCloud(*cloud, *cloud_tf_1, transform);
117  //std::cout << transform.matrix() << std::endl << std::endl;
118 
119  Eigen::Vector4f centroid (Eigen::Vector4f::Zero());
120  pcl::compute3DCentroid(*cloud, centroid);
121  Eigen::Vector4f centroid_new (Eigen::Vector4f::Zero());
122  centroid_new.head<3>() = rotation * centroid.head<3>();
123  transform.translation() = centroid.head<3>() - centroid_new.head<3>();
124 
125  pcl::transformPointCloud(*cloud, *cloud_tf_2, transform);
126  //std::cout << transform.matrix() << std::endl << std::endl;
127 
128  for (unsigned int i = 0 ; i < l_meca.size() ; i++) {
129  helper::WriteAccessor<Data <defaulttype::Vec3dTypes::VecCoord> > x = l_meca[i]->write(core::VecCoordId::position());
130  helper::WriteAccessor<Data <defaulttype::Vec3dTypes::VecCoord> > xrest = l_meca[i]->write(core::VecCoordId::restPosition());
131  helper::WriteAccessor<Data <defaulttype::Vec3dTypes::VecCoord> > xfree = l_meca[i]->write(core::VecCoordId::freePosition());
132  size_t k = 0 ;
133  for (const auto & pt : *cloud_tf_2) {
134  x[k++] = defaulttype::Vector3(pt.x, pt.y, pt.z) ;
135  }
136  for (k = 0 ; k < xfree.size() ; k++) {
137  xfree[k] = x[k] ;
138  }
139  for (k = 0 ; k < xrest.size() ; k++) {
140  xrest[k] = x[k] ;
141  }
142  }

150  void handleEvent(sofa::core::objectmodel::Event* event) override {
151  if (sofa::core::objectmodel::KeypressedEvent * ev = dynamic_cast<sofa::core::objectmodel::KeypressedEvent*>(event)){
152  if (ev->getKey() == 'T') {
153  d_tr.setValue(!d_tr.getValue()) ;
154  }
155  }
156  if (sofa::core::objectmodel::MouseEvent * ev = dynamic_cast<sofa::core::objectmodel::MouseEvent*>(event)){
157  if (m_applyInitial == true) {
158  applyTransform(d_initPhi.getValue());
159  m_applyInitial = false;
160  return;
161  }
162 
163  defaulttype::Vector3 phi (0,0,0) ;
164  static int selected_dim = -1 ; // 0 x, 1 y, 2 z
165  if (ev->getState() == core::objectmodel::MouseEvent::LeftPressed) {
166  // set x
167  x_0 = ev->getPosX() ; y_0 = ev->getPosY() ;
168  selected_dim = 0 ;
169  }
170  else if (ev->getState() == core::objectmodel::MouseEvent::RightPressed) {
171  // set y
172  x_0 = ev->getPosX() ; y_0 = ev->getPosY() ;
173  selected_dim = 1 ;
174  }
175  else if (ev->getState() == core::objectmodel::MouseEvent::MiddlePressed) {
176  // set z
177  x_0 = ev->getPosX() ; y_0 = ev->getPosY() ;
178  selected_dim = 2 ;
179  }
180  else if (ev->getState() == core::objectmodel::MouseEvent::LeftReleased ||
181  ev->getState() == core::objectmodel::MouseEvent::RightReleased ||
182  ev->getState() == core::objectmodel::MouseEvent::MiddleReleased) {
183  selected_dim = -1 ;
184  }
185  else if (ev->getState() == core::objectmodel::MouseEvent::Move) {
186  x_1 = ev->getPosX() ; y_1 = ev->getPosY() ;
187  float a = -(float)(y_1-y_0) / 15.f ;//(float)(x_1-x_0) ;
188 
189  if (selected_dim >= 0 && selected_dim <= 2) {
190  phi[selected_dim] += a/1.5f ;
191  if ((int)phi[selected_dim] > 180)
192  phi[selected_dim] = 180.f ;
193  else if ((int)phi[selected_dim] < -180)
194  phi[selected_dim] = -180.f ;
195  }
196  }
197  else if (ev->getState() == core::objectmodel::MouseEvent::Wheel){
198  std::cout << ev->getWheelDelta() << std::endl ;
199  }
200  if (!d_tr.getValue()) {
201  applyTransform(phi);
202  } else {
203  translate(phi) ;
204  }
205  }
