60 PointType bary (const PointCloud & pointset) {
61  PointType out (0., 0., 0.) ;
62  for (const auto & point : pointset) {
63  out.x = out.x + point.x ;
64  out.y = out.y + point.y ;
65  out.z = out.z + point.z ;
66  }
67  out.x = out.x / pointset.size() ;
68  out.y = out.y / pointset.size() ;
69  out.z = out.z / pointset.size() ;
70  return out ;
71 }

79 PointType stdev (const PointCloud & pointset, const PointType & mu) {
80  PointType out (0., 0., 0.) ;
81  PointType mu_sqr (
82  mu.x*mu.x,
83  mu.y*mu.y,
84  mu.z*mu.z
85  ) ;
86  for (const auto & point : pointset) {
87  out.x += (point.x - mu_sqr.x)*(point.x - mu_sqr.x) ;
88  out.y += (point.y - mu_sqr.y)*(point.y - mu_sqr.y) ;
89  out.z += (point.z - mu_sqr.z)*(point.z - mu_sqr.z) ;
90  }
91  out.x = std::sqrt(out.x / pointset.size()) ;
92  out.y = std::sqrt(out.y / pointset.size()) ;
93  out.z = std::sqrt(out.z / pointset.size()) ;
94  return out ;
95 }

145  PointCloud::Ptr out (new PointCloud) ; out->clear() ;

148  auto mu = bary (*in) ;
149  auto sigma = stdev(*in, mu) ;
150 // std::cout << mu << " " << sigma << std::endl ;

167  bool point_in_sigma_mu (const PointType & v, const PointType & sigma, const PointType & mu) {
168  double alpha = d_alpha.getValue() ;
169  PointType threshold (
170  sigma.x * alpha,
171  sigma.y * alpha,
172  sigma.z * alpha
173  ) ;
174  if (v.x >= mu.x + threshold.x || v.x <= mu.x - threshold.x ||
175  v.y >= mu.y + threshold.y || v.y <= mu.y - threshold.y ||
176  v.z >= mu.z + threshold.z || v.z <= mu.z - threshold.z) {
177  return false ;
178  }
179  return true ;
180  }
